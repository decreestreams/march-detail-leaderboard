  <script>

    // ===============================

    const data = [

      { name: "Yeraldy", sales: 4080 },

      { name: "Baily", sales: 3384 },

      { name: "Morgan", sales: 3332 },

      { name: "Gunnar", sales: 3180 },

      { name: "Abigail", sales: 2076 },

      { name: "Trevor", sales: 0 }

    ];


    // Sort descending

    data.sort((a, b) => b.sales - a.sales);


    const tbody = document.getElementById("leaderboardBody");


    data.forEach((person, index) => {

      const tr = document.createElement("tr");


      // Medal styling

      if (index === 0) tr.className = "gold-row";

      else if (index === 1) tr.className = "silver-row";

      else if (index === 2) tr.className = "bronze-row";

      else tr.className = "border-b";


      tr.innerHTML = `

        <td class="p-4 font-extrabold text-2xl">#${index + 1}</td>

        <td class="p-4 font-semibold">${person.name}</td>

        <td class="p-4 text-right font-extrabold">$${person.sales.toLocaleString()}</td>

      `;


      tbody.appendChild(tr);

    });


    // Optional: auto fullscreen when opened on TV (user must allow)

    document.addEventListener("keydown", (e) => {

      if (e.key.toLowerCase() === "f") {

        if (!document.fullscreenElement) {

          document.documentElement.requestFullscreen().catch(() => {});

        }

      }

    });

  </script>
